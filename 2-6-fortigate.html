Fortinet FortiGate is a leading Next-Generation Firewall known for its custom Security Processing Units (SPUs) that provide high performance. FortiGate is part of the Fortinet Security Fabric, integrating with other Fortinet products for unified security.

FortiGate Architecture

Fortinet differentiates itself through custom ASIC chips (Security Processing Units) that accelerate security functions in hardware, providing high performance even with all features enabled.

config vpn ipsec phase1-interface
  edit "VPN-to-Branch"
    set interface "wan1"
    set ike-version 2
    set peertype any
    set proposal aes256-sha256
    set dhgrp 14 20
    set remote-gw 203.0.113.50
    set psksecret "YourSecretKey"
    set dpd on-idle
    set dpd-retryinterval 10
  next
end

# Phase 2 Configuration
config vpn ipsec phase2-interface
  edit "VPN-to-Branch-P2"
    set phase1name "VPN-to-Branch"
    set proposal aes256-sha256
    set pfs enable
    set dhgrp 14
    set auto-negotiate enable
    set src-subnet 10.1.0.0/16
    set dst-subnet 10.2.0.0/16
  next
end

# Static Route for remote network
config router static
  edit 0
    set dst 10.2.0.0/16
    set device "VPN-to-Branch"
  next
end

# Firewall Policy for VPN traffic
config firewall policy
  edit 0
    set name "Internal-to-Branch"
    set srcintf "internal"
    set dstintf "VPN-to-Branch"
    set srcaddr "all"
    set dstaddr "all"
    set action accept
    set schedule "always"
    set service "ALL"
  next
end
