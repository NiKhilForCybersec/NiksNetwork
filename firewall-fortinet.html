FortiGate Firewall Guide

Complete guide to Fortinet FortiGate next-generation firewalls. Covers FortiOS configuration, Security Fabric, SD-WAN, VDOMs, and operational commands.

Platform Overview

FortiGate is Fortinet's flagship NGFW product line, known for custom ASIC hardware acceleration (NP/CP processors) and tight integration with the Fortinet Security Fabric.

Fortinet Security Fabric

The Security Fabric provides integrated security across Fortinet products with centralized management, automated threat response, and shared threat intelligence.

FortiManager

Centralized management, policy deployment, and configuration backup for multiple FortiGates.

FortiAnalyzer

Log aggregation, reporting, SIEM functionality, and compliance reporting.

FortiSandbox

Advanced threat protection with file sandboxing for zero-day detection.

FortiClient EMS

Endpoint management, VPN deployment, and endpoint telemetry integration.

FortiSwitch

Fabric-integrated switching with security policy enforcement at the access layer.

FortiAP

Wireless access points managed directly from FortiGate with integrated security.

Firewall Policy Configuration

FortiOS Firewall Policy Examples 
# =============================================================================
# FORTIGATE FIREWALL POLICY CONFIGURATION
# =============================================================================

# Basic outbound policy - LAN to Internet
config firewall policy
    edit 1
        set name "LAN-to-Internet"
        set srcintf "lan"
        set dstintf "wan1"
        set srcaddr "LAN-Network"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set utm-status enable
        set ssl-ssh-profile "certificate-inspection"
        set av-profile "default"
        set webfilter-profile "default"
        set ips-sensor "default"
        set application-list "default"
        set logtraffic all
        set nat enable
    next
end

# Inbound policy - Internet to DMZ Web Server
config firewall policy
    edit 2
        set name "Internet-to-WebServer"
        set srcintf "wan1"
        set dstintf "dmz"
        set srcaddr "all"
        set dstaddr "WebServer-VIP"
        set action accept
        set schedule "always"
        set service "HTTP" "HTTPS"
        set utm-status enable
        set ssl-ssh-profile "deep-inspection"
        set av-profile "high-security"
        set ips-sensor "protect-servers"
        set waf-profile "web-protection"
        set logtraffic all
        set nat disable
    next
end

# Inter-VLAN policy with application control
config firewall policy
    edit 3
        set name "Users-to-Servers"
        set srcintf "user-vlan"
        set dstintf "server-vlan"
        set srcaddr "User-Subnet"
        set dstaddr "Server-Subnet"
        set action accept
        set schedule "business-hours"
        set service "ALL"
        set groups "Domain-Users"
        set utm-status enable
        set application-list "business-apps"
        set logtraffic all
        set nat disable
    next
end

# Deny policy with logging
config firewall policy
    edit 100
        set name "Deny-All"
        set srcintf "any"
        set dstintf "any"
        set srcaddr "all"
        set dstaddr "all"
        set action deny
        set schedule "always"
        set service "ALL"
        set logtraffic all
    next
end

NAT Configuration

FortiOS NAT and VIP Configuration 
# =============================================================================
# NAT CONFIGURATION
# =============================================================================

# Virtual IP (DNAT) - Port Forward to Web Server
config firewall vip
    edit "WebServer-VIP"
        set extip 203.0.113.100
        set mappedip "192.168.10.100"
        set extintf "wan1"
        set portforward enable
        set extport 443
        set mappedport 443
    next
end

# VIP with multiple ports
config firewall vip
    edit "MailServer-VIP"
        set extip 203.0.113.101
        set mappedip "192.168.10.101"
        set extintf "wan1"
        set portforward enable
        set protocol tcp
        set extport 25
        set mappedport 25
    next
    edit "MailServer-VIP-IMAP"
        set extip 203.0.113.101
        set mappedip "192.168.10.101"
        set extintf "wan1"
        set portforward enable
        set protocol tcp
        set extport 993
        set mappedport 993
    next
end

# IP Pool (SNAT) - Source NAT with IP pool
config firewall ippool
    edit "SNAT-Pool"
        set startip 203.0.113.50
        set endip 203.0.113.60
        set type overload
    next
end

# Central NAT (separate from policy)
config firewall central-snat-map
    edit 1
        set srcintf "lan"
        set dstintf "wan1"
        set orig-addr "LAN-Network"
        set dst-addr "all"
        set nat-ippool "SNAT-Pool"
    next
end

Security Profiles (UTM)

FortiOS Security Profile Configuration 
# =============================================================================
# SECURITY PROFILES
# =============================================================================

# Antivirus Profile
config antivirus profile
    edit "strict-av"
        set comment "Strict AV scanning"
        config http
            set av-scan block
            set outbreak-prevention block
        end
        config ftp
            set av-scan block
        end
        config smtp
            set av-scan block
        end
        set analytics-db enable
        set mobile-malware-db enable
    next
end

# Web Filter Profile
config webfilter profile
    edit "corporate-webfilter"
        set comment "Corporate web filtering policy"
        config ftgd-wf
            config filters
                edit 1
                    set category 1  # Drug Abuse
                    set action block
                next
                edit 2
                    set category 3  # Hacking
                    set action block
                next
                edit 3
                    set category 4  # Illegal/Unethical
                    set action block
                next
                edit 4
                    set category 14  # Pornography
                    set action block
                next
                edit 5
                    set category 26  # Malicious Websites
                    set action block
                next
                edit 6
                    set category 88  # Newly Observed Domain
                    set action warning
                next
            end
        end
        set web-content-log enable
        set web-filter-activex-log enable
        set web-filter-js-log enable
    next
end

# IPS Sensor
config ips sensor
    edit "protect-servers"
        set comment "IPS for server protection"
        config entries
            edit 1
                set severity high critical
                set action block
                set status enable
            next
            edit 2
                set severity medium
                set action default
                set status enable
            next
        end
        config filter
            edit 1
                set severity high critical
                set action block
            next
        end
    next
end

# Application Control
config application list
    edit "business-apps"
        set comment "Allow business applications"
        config entries
            edit 1
                set category 5  # Business
                set action pass
            next
            edit 2
                set category 17  # Social Media
                set action block
            next
            edit 3
                set category 25  # P2P
                set action block
            next
            edit 4
                set category 6  # Cloud IT
                set action pass
            next
        end
    next
end

# SSL/SSH Inspection
config firewall ssl-ssh-profile
    edit "deep-inspection"
        set comment "Full SSL inspection"
        config ssl
            set inspect-all deep-inspection
        end
        config https
            set ports 443
            set status deep-inspection
        end
        config ftps
            set status deep-inspection
        end
        config imaps
            set status deep-inspection
        end
        set server-cert-mode re-sign
        set caname "Fortinet_CA_SSL"
    next
end

VPN Configuration

IPSec Site-to-Site VPN

FortiOS IPSec VPN Configuration 
# =============================================================================
# IPSEC SITE-TO-SITE VPN
# =============================================================================

# Phase 1 (IKE)
config vpn ipsec phase1-interface
    edit "Site-B-VPN"
        set interface "wan1"
        set ike-version 2
        set peertype any
        set net-device disable
        set proposal aes256-sha256
        set dhgrp 14
        set remote-gw 203.0.113.2
        set psksecret "YourSecretKey123!"
        set dpd on-idle
        set dpd-retrycount 3
        set dpd-retryinterval 10
    next
end

# Phase 2 (IPSec)
config vpn ipsec phase2-interface
    edit "Site-B-VPN-P2"
        set phase1name "Site-B-VPN"
        set proposal aes256-sha256
        set dhgrp 14
        set auto-negotiate enable
        set src-subnet 10.1.0.0 255.255.0.0
        set dst-subnet 10.2.0.0 255.255.0.0
    next
end

# Static Route for VPN traffic
config router static
    edit 10
        set dst 10.2.0.0 255.255.0.0
        set device "Site-B-VPN"
    next
end

# Firewall Policy for VPN
config firewall policy
    edit 10
        set name "LAN-to-Site-B"
        set srcintf "lan"
        set dstintf "Site-B-VPN"
        set srcaddr "LAN-Subnet"
        set dstaddr "Site-B-Subnet"
        set action accept
        set schedule "always"
        set service "ALL"
        set logtraffic all
    next
    edit 11
        set name "Site-B-to-LAN"
        set srcintf "Site-B-VPN"
        set dstintf "lan"
        set srcaddr "Site-B-Subnet"
        set dstaddr "LAN-Subnet"
        set action accept
        set schedule "always"
        set service "ALL"
        set logtraffic all
    next
end

SD-WAN Configuration

FortiOS SD-WAN Configuration 
# =============================================================================
# SD-WAN CONFIGURATION
# =============================================================================

# SD-WAN Zone
config system sdwan
    set status enable
    
    # Define SD-WAN members (WAN interfaces)
    config members
        edit 1
            set interface "wan1"
            set zone "virtual-wan-link"
            set gateway 203.0.113.1
            set cost 0
        next
        edit 2
            set interface "wan2"
            set zone "virtual-wan-link"
            set gateway 198.51.100.1
            set cost 10
        next
    end
    
    # Health Check (SLA)
    config health-check
        edit "Internet-SLA"
            set server "8.8.8.8" "1.1.1.1"
            set protocol ping
            set interval 500
            set failtime 5
            set recoverytime 5
            config sla
                edit 1
                    set latency-threshold 100
                    set jitter-threshold 50
                    set packetloss-threshold 5
                next
            end
            set members 1 2
        next
        edit "O365-SLA"
            set server "outlook.office365.com"
            set protocol http
            set port 443
            set interval 1000
            config sla
                edit 1
                    set latency-threshold 150
                next
            end
            set members 1 2
        next
    end
    
    # SD-WAN Rules
    config service
        edit 1
            set name "O365-Traffic"
            set mode sla
            set dst "Microsoft-Office365"
            config sla
                edit "O365-SLA"
                    set id 1
                next
            end
            set priority-members 1 2
        next
        edit 2
            set name "VoIP-Traffic"
            set mode priority
            set protocol 17
            set start-port 5060
            set end-port 5061
            set priority-members 1
        next
        edit 3
            set name "Default-Traffic"
            set mode load-balance
            set dst "all"
            set load-balance-mode source-ip-based
        next
    end
end

# Firewall Policy referencing SD-WAN zone
config firewall policy
    edit 1
        set name "LAN-to-SDWAN"
        set srcintf "lan"
        set dstintf "virtual-wan-link"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set nat enable
    next
end

Virtual Domains (VDOMs)

VDOMs allow a single FortiGate to be partitioned into multiple virtual firewalls, each with independent routing, policies, and administrators.

FortiOS VDOM Configuration 
# =============================================================================
# VDOM CONFIGURATION
# =============================================================================

# Enable VDOMs
config system global
    set vdom-mode multi-vdom
end

# Create VDOM
config vdom
    edit "Customer-A"
    next
    edit "Customer-B"
    next
end

# Assign interfaces to VDOM
config global
    config system interface
        edit "port3"
            set vdom "Customer-A"
            set ip 192.168.100.1 255.255.255.0
            set allowaccess ping https ssh
        next
        edit "port4"
            set vdom "Customer-B"
            set ip 192.168.200.1 255.255.255.0
            set allowaccess ping https ssh
        next
    end
end

# Configure VDOM-specific settings
config vdom
    edit "Customer-A"
        config system settings
            set opmode nat
        end
        config router static
            edit 1
                set gateway 10.0.0.1
                set device "port3"
            next
        end
        config firewall policy
            edit 1
                set name "Outbound"
                set srcintf "port3"
                set dstintf "wan-link"
                set srcaddr "all"
                set dstaddr "all"
                set action accept
                set schedule "always"
                set service "ALL"
                set nat enable
            next
        end
    next
end

# Inter-VDOM Link
config global
    config system vdom-link
        edit "vlink1"
        next
    end
    config system interface
        edit "vlink1_0"
            set vdom "Customer-A"
            set ip 169.254.1.1 255.255.255.252
        next
        edit "vlink1_1"
            set vdom "Customer-B"
            set ip 169.254.1.2 255.255.255.252
        next
    end
end

Essential CLI Commands

FortiOS Operational Commands 
# =============================================================================
# SYSTEM COMMANDS
# =============================================================================
get system status                    # System overview
get system performance status        # CPU, memory, sessions

get hardware nic                     # NIC details
execute reboot                       # Reboot device
execute shutdown                     # Shutdown device

# =============================================================================
# CONFIGURATION COMMANDS
# =============================================================================
show                                 # Show current context config
show full-configuration              # All config
get                                  # Get current values
config system interface              # Enter config context
    edit "port1"                     # Edit specific item
    set ip 192.168.1.1/24           # Set value
    end                              # Exit and save
execute backup config ftp    # Backup config
execute restore config ftp   # Restore config

# =============================================================================
# NETWORK COMMANDS
# =============================================================================
get router info routing-table all    # Routing table
get router info ospf neighbor        # OSPF neighbors
get router info bgp summary          # BGP summary
execute ping 8.8.8.8                 # Ping
execute traceroute 8.8.8.8           # Traceroute
diagnose ip arp list                 # ARP table

# =============================================================================
# FIREWALL/SESSION COMMANDS
# =============================================================================
diagnose sys session list            # All sessions
diagnose sys session filter src 192.168.1.10
diagnose sys session filter clear    # Clear filters
diagnose sys session stat            # Session statistics
get firewall policy                  # Show policies
diagnose firewall policy list        # Policy hit counts

# VPN COMMANDS
# =============================================================================
diagnose vpn ike gateway list        # IKE SAs
diagnose vpn tunnel list             # IPSec SAs
get vpn ipsec tunnel summary         # Tunnel summary
diagnose vpn ssl statistics          # SSL VPN stats
get vpn ssl monitor                  # SSL VPN users
execute vpn sslvpn del-tunnel  # Disconnect user

# =============================================================================
# SD-WAN COMMANDS
# =============================================================================
diagnose sys sdwan health-check      # Health check status
diagnose sys sdwan member            # Member status
diagnose sys sdwan service           # Service rules

# =============================================================================
# DEBUGGING
# =============================================================================
diagnose debug enable
diagnose debug flow filter addr 192.168.1.10
diagnose debug flow trace start 100  # Trace packets
diagnose debug flow trace stop
diagnose debug disable               # Always disable after!

# =============================================================================
# LOG COMMANDS
# =============================================================================
execute log filter category traffic
execute log filter start-line 1
execute log display                  # View logs
diagnose log test                    # Generate test log
